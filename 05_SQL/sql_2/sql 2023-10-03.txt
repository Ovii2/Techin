SELECT * 
FROM DARBUOTOJAI
WHERE GIMIMOMETAI  BETWEEN '1960-01-01' AND '1980-12-31';

SELECT * 
FROM DARBUOTOJAI
WHERE PAVARDE  BETWEEN 'B' AND 'Mz';

SELECT * 
FROM DARBUOTOJAI
WHERE PAREIGOS  IN ('Programuotojas', 'Programuotoja', 'Testuotoja', 'Testuotojas');

SELECT * 
FROM DARBUOTOJAI
WHERE PAREIGOS LIKE '%uotoj%';

SELECT * 
FROM DARBUOTOJAI
WHERE DIRBANUO LIKE '%-04-%';

SELECT * 
FROM DARBUOTOJAI
WHERE DIRBANUO LIKE '____-04-__';

SELECT * FROM DARBUOTOJAI
WHERE ASMENSKODAS IS NULL;

SELECT * FROM DARBUOTOJAI
WHERE NOT ASMENSKODAS IS NULL;

SELECT * FROM DARBUOTOJAI
ORDER BY VARDAS, PAVARDE  

SELECT * FROM DARBUOTOJAI
WHERE VARDAS ILIKE 'Mantas';

SELECT LOWER (VARDAS), UPPER(PAVARDE) 
FROM DARBUOTOJAI 

SELECT LOWER (VARDAS), SUBSTR(PAVARDE,1) 
FROM DARBUOTOJAI 

SELECT CONCAT(LOWER(VARDAS),' - ', UPPER(PAVARDE)) AS `VARDAS, PAVARDE`, CONCAT(SUBSTR(VARDAS,1,1), LOWER(SUBSTR(VARDAS,2,50))) AS VARDAS
FROM DARBUOTOJAI;

SELECT PROJEKTONUMERIS/3 FROM DARBUOTOJAI 

SELECT PROJEKTONUMERIS, COUNT(*)
FROM DARBUOTOJAI
GROUP BY PROJEKTONUMERIS;

SELECT PROJEKTONUMERIS, PAREIGOS, COUNT(*)
FROM DARBUOTOJAI
GROUP BY PROJEKTONUMERIS, PAREIGOS;

SELECT PROJEKTONUMERIS, PAREIGOS, COUNT(*)
FROM DARBUOTOJAI
GROUP BY PROJEKTONUMERIS, PAREIGOS
HAVING COUNT(*) = 1

SELECT PROJEKTONUMERIS, PAREIGOS, COUNT(*)
FROM DARBUOTOJAI
WHERE PROJEKTONUMERIS IN (1,3)
GROUP BY PROJEKTONUMERIS, PAREIGOS
HAVING COUNT(*) = 1
ORDER BY PAREIGOS DESC
LIMIT 4;
